# خريطة لتخزين رسائل الفيديو مؤقتاً في الذاكرة لتوليد الروابط
file_db = {}

@bot.on_message(filters.private & (filters.video | filters.document))
async def handle_video(client, message):
    # التحقق من أنك أنت الأدمن فقط من تستخدم البوت لحماية باقة السيرفر
    if message.chat.id != ADMIN_CHAT_ID:
        await message.reply_text("عذراً، هذا البوت خاص بالمطور فقط.")
        return

    file_id = message.id
    file_db[file_id] = message
    
    # توليد الرابط العام المباشر من سيرفر Koyeb
    public_link = f"http://{DOMAIN}/stream/{file_id}"
    
    await message.reply_text(
        f"✅ **تم توليد الرابط العام بنجاح!**\n\n"
        f"🔗 **رابط المشاهدة والتحميل:**\n`{public_link}`\n\n"
        f"ℹ️ هذا الرابط يدعم الأحجام الكبيرة والبث المباشر (Stream)."
    )

# دالة ذكية لضخ الفيديو للمتصفح مباشرة دون تحميله على سيرفر Koyeb
async def media_streamer(message):
    async for chunk in bot.download_media(message, block=True):
        yield chunk

@app.get("/stream/{file_id}")
async def stream_file(file_id: int):
    if file_id not in file_db:
        return {"error": "الملف غير موجود أو انتهت صلاحيته"}
    
    message = file_db[file_id]
    media = message.video or message.document
    
    return StreamingResponse(
        media_streamer(message),
        media_type=media.mime_type,
        headers={
            "Content-Disposition": f"attachment; filename={media.file_name}",
            "Content-Length": str(media.file_size)
        }
    )

# تشغيل البوت والخادم معاً
async def start_services():
    await bot.start()
    config = uvicorn.Config(app, host="0.0.0.0", port=PORT, log_level="info")
    server = uvicorn.Server(config)
    await server.serve()

if __name__ == "__main__":
    asyncio.run(start_services())
